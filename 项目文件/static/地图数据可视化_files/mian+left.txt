function MouseWheelHandler(e) {
    // cross-browser wheel delta
    console.log("识别到鼠标滚动");
    var e = window.event || e; // old IE support
    var delta = Math.max(-1, Math.min(1, (e.wheelDelta || -e.detail)));
    range=range+5*delta;
 }
